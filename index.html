const SCRIPT_URL = 'PASTE_YOUR_WEB_APP_URL_HERE'; const formData = new FormData(document.getElementById('registrationForm')); const ref = 'GRD-' + new Date().getFullYear() + '-' + Math.random().toString(36).substring(2,7).toUpperCase(); const adults = []; document.querySelectorAll('[name^="adult_first_"]').forEach((el, i) => { const n = i + 1; const first = el.value.trim(); const last = document.querySelector(`[name="adult_last_${n}"]`)?.value.trim(); const phone = document.querySelector(`[name="adult_phone_${n}"]`)?.value.trim(); const rel = document.querySelector(`[name="adult_rel_${n}"]`)?.value.trim(); if (first) adults.push(`${first} ${last} (${rel}) - ${phone}`); }); const children = []; document.querySelectorAll('[name^="child_first_"]').forEach((el, i) => { const n = i + 1; const first = el.value.trim(); const last = document.querySelector(`[name="child_last_${n}"]`)?.value.trim(); const age = document.querySelector(`[name="child_age_${n}"]`)?.value.trim(); const school = document.querySelector(`[name="child_school_${n}"]`)?.value.trim(); if (first) children.push(`${first} ${last}, Age ${age}, ${school}`); }); const payload = { timestamp: new Date().toLocaleString('en-KE'), ref, hof_first: formData.get('hof_first'), hof_last: formData.get('hof_last'), hof_role: formData.get('hof_role'), phone_primary: formData.get('phone_primary'), phone_alt: formData.get('phone_alt'), email_primary: formData.get('email_primary'), addr_estate: formData.get('addr_estate'), addr_road: formData.get('addr_road'), addr_town: formData.get('addr_town'), addr_county: formData.get('addr_county'), addr_postcode: formData.get('addr_postcode'), addr_landmark: formData.get('addr_landmark'), spouse: `${formData.get('sp_first')} ${formData.get('sp_last')}`.trim(), spouse_phone: formData.get('sp_phone'), other_adults: adults.join(' | '), children: children.join(' | '), notes: formData.get('notes') }; fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify(payload) }) .then(() => { document.getElementById('refCode').textContent = 'Reference: ' + ref; document.getElementById('registrationForm').style.display = 'none'; document.querySelector('.notice').style.display = 'none'; document.getElementById('successScreen').classList.add('show'); document.getElementById('progressBar').style.width = '100%'; window.scrollTo({ top: 0, behavior: 'smooth' }); }) .catch(() => { btn.disabled = false; text.textContent = 'Submit family registration'; spinner.style.display = 'none'; alert('Submission failed. Please check your connection and try again.'); });
